## Overview

The CheckboxGroup component provides a container for managing multiple checkboxes with coordinated state. It handles the selection of multiple values, propagates disabled and invalid states to child checkboxes, and supports both horizontal and vertical layouts.

## Installation

```bash
npm install @shizen-ui/svelte
```

## Import

```typescript
import { CheckboxGroup, Checkbox } from '@shizen-ui/svelte';
```

## Props

<ParamField path="value" type="string[]" default={[]}>
  Array of selected checkbox values. Use `bind:value` for two-way binding.
</ParamField>

<ParamField path="name" type="string">
  The name attribute shared by all checkboxes in the group. Auto-generated if not provided.
</ParamField>

<ParamField path="orientation" type="string" default="vertical">
  The layout direction of the checkbox group.
  
  Options: `"vertical"` | `"horizontal"`
</ParamField>

<ParamField path="disabled" type="boolean" default={false}>
  When true, disables all checkboxes in the group.
</ParamField>

<ParamField path="invalid" type="boolean" default={false}>
  When true, marks all checkboxes in the group as invalid for error styling.
</ParamField>

<ParamField path="required" type="boolean" default={false}>
  When true, indicates the group requires at least one selection.
</ParamField>

<ParamField path="id" type="string">
  The ID for the checkbox group container. Auto-generated if not provided.
</ParamField>

<ParamField path="class" type="string">
  Additional CSS classes to apply to the group container.
</ParamField>

## Validation

### Required Group

```svelte
<script>
  import { CheckboxGroup, Checkbox, TextField, Label, FieldError } from '@shizen-ui/svelte';
  
  let skills = $state<string[]>([]);
  let showError = $state(false);
  
  function validate() {
    showError = skills.length === 0;
  }
</script>

<TextField invalid={showError}>
  <Label>Select your skills (required)</Label>
  <CheckboxGroup bind:value={skills} required invalid={showError}>
    <Checkbox value="html">
      <Checkbox.Control />
      <Checkbox.Content>HTML</Checkbox.Content>
    </Checkbox>
    
    <Checkbox value="css">
      <Checkbox.Control />
      <Checkbox.Content>CSS</Checkbox.Content>
    </Checkbox>
    
    <Checkbox value="javascript">
      <Checkbox.Control />
      <Checkbox.Content>JavaScript</Checkbox.Content>
    </Checkbox>
  </CheckboxGroup>
  {#if showError}
    <FieldError>Please select at least one skill</FieldError>
  {/if}
</TextField>

<button onclick={validate}>Validate</button>
```

## Context Propagation

The CheckboxGroup automatically provides context to child Checkbox components:

- `value` - The array of selected values
- `name` - Shared name for all checkboxes
- `disabled` - Disabled state inherited by all children
- `invalid` - Invalid state inherited by all children
- `orientation` - Layout direction
- `toggleValue(value: string)` - Function to toggle a value in the selection

Child checkboxes automatically coordinate with the group and don't need individual `bind:checked` props.

<Note>
  When a Checkbox is inside a CheckboxGroup, its `value` prop is required to identify which option it represents in the group's value array.
</Note>

## Accessibility

- Uses `role="group"` for semantic grouping of checkboxes
- Child checkboxes maintain full keyboard navigation
- Group state (`disabled`, `invalid`) is propagated via data attributes
- Data attributes applied:
  - `data-disabled` - When the group is disabled
  - `data-invalid` - When the group is invalid
  - `data-orientation` - The layout orientation
- Combine with Label component for proper labeling:

```svelte
<TextField>
  <Label>Choose your preferences</Label>
  <CheckboxGroup bind:value={selected}>
    <!-- checkboxes -->
  </CheckboxGroup>
</TextField>
```

- Use Description for additional context
- Use FieldError to display validation messages

<Tip>
  Use CheckboxGroup with TextField, Label, and FieldError components for complete form field functionality with proper accessibility and validation.
</Tip>

## Form Integration

All checkboxes in the group share the same `name` attribute, allowing multiple values to be submitted with the form:

```svelte
<script>
  import { CheckboxGroup, Checkbox } from '@shizen-ui/svelte';
  
  let selected = $state<string[]>(['option1']);
  
  function handleSubmit(e: SubmitEvent) {
    e.preventDefault();
    const formData = new FormData(e.target as HTMLFormElement);
    const values = formData.getAll('preferences');
    console.log('Selected:', values);
  }
</script>

<form onsubmit={handleSubmit}>
  <CheckboxGroup bind:value={selected} name="preferences">
    <Checkbox value="option1">
      <Checkbox.Control />
      <Checkbox.Content>Option 1</Checkbox.Content>
    </Checkbox>
    
    <Checkbox value="option2">
      <Checkbox.Control />
      <Checkbox.Content>Option 2</Checkbox.Content>
    </Checkbox>
  </CheckboxGroup>
  
  <button type="submit">Submit</button>
</form>
```
