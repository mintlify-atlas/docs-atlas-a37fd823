## Overview

The RadioGroup component provides a container for managing radio buttons with coordinated single-selection behavior. It ensures only one radio can be selected at a time, propagates disabled and invalid states to child radios, and supports both horizontal and vertical layouts with arrow key navigation.

## Installation

```bash
npm install @shizen-ui/svelte
```

## Import

```typescript
import { RadioGroup, Radio } from '@shizen-ui/svelte';
```

## Props

<ParamField path="value" type="string">
  The value of the currently selected radio button. Use `bind:value` for two-way binding.
</ParamField>

<ParamField path="name" type="string">
  The name attribute shared by all radio buttons in the group. Auto-generated if not provided.
</ParamField>

<ParamField path="orientation" type="string" default="vertical">
  The layout direction of the radio group.
  
  Options: `"vertical"` | `"horizontal"`
</ParamField>

<ParamField path="disabled" type="boolean" default={false}>
  When true, disables all radio buttons in the group.
</ParamField>

<ParamField path="invalid" type="boolean" default={false}>
  When true, marks all radio buttons in the group as invalid for error styling.
</ParamField>

<ParamField path="required" type="boolean" default={false}>
  When true, indicates the group requires a selection.
</ParamField>

<ParamField path="id" type="string">
  The ID for the radio group container. Auto-generated if not provided.
</ParamField>

<ParamField path="class" type="string">
  Additional CSS classes to apply to the group container.
</ParamField>

## Validation

### Required Group

```svelte
<script>
  import { RadioGroup, Radio, TextField, Label, FieldError } from '@shizen-ui/svelte';
  
  let plan = $state<string>();
  let showError = $state(false);
  
  function validate() {
    showError = !plan;
  }
</script>

<TextField invalid={showError}>
  <Label>Select a plan (required)</Label>
  <RadioGroup bind:value={plan} required invalid={showError}>
    <Radio value="free">
      <Radio.Control />
      <Radio.Content>Free</Radio.Content>
    </Radio>
    
    <Radio value="basic">
      <Radio.Control />
      <Radio.Content>Basic - $10/mo</Radio.Content>
    </Radio>
    
    <Radio value="pro">
      <Radio.Control />
      <Radio.Content>Pro - $25/mo</Radio.Content>
    </Radio>
  </RadioGroup>
  {#if showError}
    <FieldError>Please select a plan</FieldError>
  {/if}
</TextField>

<button onclick={validate}>Validate</button>
```

## Context Propagation

The RadioGroup automatically provides context to child Radio components:

- `value` - The currently selected value
- `name` - Shared name for all radio buttons
- `disabled` - Disabled state inherited by all children
- `invalid` - Invalid state inherited by all children
- `orientation` - Layout direction
- `setValue(value: string)` - Function to select a radio

Child radios automatically coordinate with the group and only need their `value` prop set.

<Note>
  When a Radio is inside a RadioGroup, its `value` prop is required to identify which option it represents. The `checked` prop is automatically managed by the group.
</Note>

## Accessibility

- Uses `role="radiogroup"` for semantic grouping
- Proper keyboard navigation:
  - **Tab** focuses the checked radio (or first if none checked)
  - **Arrow Up/Down** (vertical) or **Arrow Left/Right** (horizontal) to navigate between options
  - **Space** or **Enter** to select focused radio
- Only the checked radio (or first radio if none checked) has `tabindex="0"`, others have `tabindex="-1"`
- Group state propagated via data attributes:
  - `data-disabled` - When the group is disabled
  - `data-invalid` - When the group is invalid
  - `data-orientation` - The layout orientation
- All radios share the same `name` attribute for proper form behavior
- Combine with Label component for proper labeling:

```svelte
<TextField>
  <Label for="size-group">Choose your size</Label>
  <RadioGroup id="size-group" bind:value={selected}>
    <!-- radios -->
  </RadioGroup>
</TextField>
```

- Use Description for additional context
- Use FieldError to display validation messages

<Tip>
  Use RadioGroup with TextField, Label, and FieldError components for complete form field functionality with proper accessibility and validation.
</Tip>

## Form Integration

All radios in the group share the same `name` attribute, allowing a single value to be submitted with the form:

```svelte
<script>
  import { RadioGroup, Radio } from '@shizen-ui/svelte';
  
  let selected = $state('option1');
  
  function handleSubmit(e: SubmitEvent) {
    e.preventDefault();
    const formData = new FormData(e.target as HTMLFormElement);
    const value = formData.get('choice');
    console.log('Selected:', value);
  }
</script>

<form onsubmit={handleSubmit}>
  <RadioGroup bind:value={selected} name="choice">
    <Radio value="option1">
      <Radio.Control />
      <Radio.Content>Option 1</Radio.Content>
    </Radio>
    
    <Radio value="option2">
      <Radio.Control />
      <Radio.Content>Option 2</Radio.Content>
    </Radio>
    
    <Radio value="option3">
      <Radio.Control />
      <Radio.Content>Option 3</Radio.Content>
    </Radio>
  </RadioGroup>
  
  <button type="submit">Submit</button>
</form>
```
